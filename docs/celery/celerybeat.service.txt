# /etc/systemd/system/celerybeat.service

[Unit]
Description=Celery Beat Service
After=network.target

[Service]
Type=simple
User=celery
Group=celery

EnvironmentFile=/etc/default/celeryd
WorkingDirectory=/home/username/parentdir/project

ExecStart=/home/username/parentdir/project/venv/bin/celery -A ${CELERY_APP} beat \
  --pidfile=${CELERYBEAT_PID_FILE} \
  --logfile=${CELERYBEAT_LOG_FILE} --loglevel=${CELERYD_LOG_LEVEL}
ExecStop=/usr/bin/pkill -f 'celery -A ${CELERY_APP} beat'
# ExecStop=/bin/systemctl kill celerybeat.service
RemainAfterExit=yes

[Install]
WantedBy=multi-user.target
